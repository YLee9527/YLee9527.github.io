# 类(基础)

# ES5 类的使用

ES6 之前，使用构造函数创建实例对象

function Point(x, y) {
  // this指向创建的实例
  this.x = x
  this.y = y
}
// 在Point的原型对象上定义一个方法
Point.prototype.getPosition = function() {
  return '(' + this.x + ',' + this.y + ')'
}
// 使用构造函数
var p1 = new Point(2, 3)
console.log(p1) // Point {x: 2, y: 3}
console.log(p1.getPosition()) // (2, 3)

# ES6 中类的基本使用

class Point2 {
  // 每一个类的定义，必须要有构造函数，他的默认返回值就是this（实例对象），也可以自己return
  constructor(x, y) {
    this.x = x
    this.y = y
  }
  getPosition() {
    return `(${this.x},${this.y})`
  }
}
const p2 = new Point2(4, 5)
console.log(p2) // Point2 {x: 2, y: 3}
console.log(p2.getPosition()) // (4, 5)
// 判断实例自身是否有某属性
console.log(p2.hasOwnProperty('x')) // true
// 实例p2的方法getPosition是它继承来的，实际上该方法是在类上
console.log(p2.__proto__.hasOwnProperty('getPosition')) // true

# 存值函数和取值函数

把一个属性拆开成两个存取值函数（set xx, get xx）,当要读取某值的时候会调用 get 修饰的方法，赋值时调用 set 修饰的方法

var info = {
  _age: 18,
  set age(newValue) {
    if (newValue > 18) {
      console.log('怎么变老了')
    } else {
      console.log('我还年轻呢')
    }
  },
  get age() {
    console.log('问我年龄干嘛？')
    return this._age
  }
}
info.age // 问我年龄干嘛？
console.log(info.age)
// 问我年龄干嘛？
// 18
info.age = 16 // 我还年轻呢
info.age = 20 // 怎么变老了
// ES6类中使用
class Info {
  constructor(age) {
    this._age = age
  }
  set age(newAge) {
    console.log('newAge is: ' + newAge)
    this._age = newAge
  }
  get age() {
    return this._age
  }
}
const info2 = new Info(13)
console.log(info2.age) // 13
info2.age = 17 // newAge is: 17
console.log(info2.age) // 17

# 静态方法与静态属性

定义类的时候，直接定义在类里面的方法都会被实例继承,如果不希望实例继承某个方法，只想让方法被类本身调用时，就需要将方法标记为静态方法（使用 static 标识）

// 静态方法
class Point3 {
  constructor(x, y) {
    this.x = x
    this.y = y
  }
  getPosition() {
    return `(${this.x},${this.y})`
  }
  static getClassName() {
    return Point3.name
  }
}
const p3 = new Point3(7, 8)
console.log(p3.getPosition()) // (7, 8)
console.log(Point3.getClassName()) // Point3
// console.log(p3.getClassName()) // Uncaught TypeError: p3.getClassName is not a function

// 静态属性
//  ES6只有静态方法，没有静态属性，可以通过如下方法
Point3.z = 7
console.log(p3.z) // undefined

# 私有方法与私有属性

有些封装的方法或插件中，有些方法只希望在内部使用，不希望暴露给使用者，这时候就需要将方法变成私有方法,

// 私有方法
// 目前ES6并不提供私有方法和私有属性，所以需要使用一些技巧实现
// 一，从命名上
class Point4 {
  // 以下划线开头的通常是私有方法，告诉使用这个类的人，不使用它（只是从名字上区分）
  _func() {}
}
// 二，将方法挪出去
const _func = () => {}
class Point5 {
  func() {
    _func.call(this)
  }
}

# new.target

用于打印当前的类

class Parent {
  constructor() {
    console.log(new.target) // 打印当前类
    if (new.target === Parent) {
      throw new Error('不能实例化')
    } else {
      console.log('通过继承实例化')
    }
  }
}
class Child extends Parent {
  constructor() {
    super()
  }
}
const c = new Child() // 通过继承实例化
const d = new Parent() // Uncaught Error: 不能实例化
