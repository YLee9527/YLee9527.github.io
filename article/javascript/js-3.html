# 类(进阶)

# ES5 中类型继承

ES5 中实现一个函数的继承是需要修改原型链的

function Food() {
  this.type = 'food'
}
Food.prototype.getType = function() {
  return this.type
}
function Vagetable(name) {
  this.name = name
}
// 这样Food创建的实例就会赋给Vagetable的原型对象
Vagetable.prototype = new Food()
const tomato = new Vagetable('tomato')
console.log(tomato.type) // food
console.log(tomato.name) // tomato

# ES6 类的继承

// ES6类的继承
class Parent {
  constructor(name) {
    this.name = name
  }
  getName() {
    return this.name
  }
}
class Child extends Parent {
  constructor(name, age) {
    // 给super的传参就相当于父类constructor的传参
    // 只有使用了super后才能使用this
    super(name)
    this.age = age
  }
}
const child = new Child('terry', 20)
console.log(child.getName()) // terry
console.log(child.age) // 20

# 实例归属

// 实例归属
console.log(child instanceof Child) // true
console.log(child instanceof Parent) // true
// 继承了父类的子类创建的实例即是父类又是子类的实例
// 字类也会继承父类的静态方法变为自己的静态方法

// Object.getPrototypeOf() 能够获取一个构造函数的原型对象
console.log(Object.getPrototypeOf(Child) === Parent) // true

# super

super 既可以作为函数也可以作为对象使用

  • super 作为函数使用，代表父类的构造函数 constructor,ES6 要求子类的构造函数中必须调用一次 super 函数,字类中添加了 super 函数，那么父类构造函数中的 this 的值也会添加到子类中
  • super 作为对象,在普通方法中他指向的父类的原型对象，在静态方法中指向的是父类
class Parent2 {
  constructor() {
    this.type = 'parent'
  }
  getName() {
    return this.type
  }
}
Parent2.getType = () => {
  return 'is Parent'
}
class Child2 extends Parent2 {
  constructor() {
    super()
    console.log('constructor: ' + super.getName())
  }
  getParentName() {
    console.log('getParentName: ' + super.getName())
  }
  getParentType() {
    console.log('getParentType: ' + super.getType())
  }
  static getParentTypeB() {
    console.log('getParentType: ' + super.getType())
  }
}
const child2 = new Child2() // constructor: parent
child2.getParentName() // getParentName: parent
// child2.getParentType() // error; 因为super指代的是父类的原型对象而不是父类本身
Child2.getParentTypeB() // getParentType: is Parent; super在静态方法中指向父类本身

class Parent3 {
  constructor() {
    this.name = 'parent'
  }
  print() {
    console.log(this.name)
  }
}
class Child3 extends Parent3 {
  constructor() {
    super()
    this.name = 'child'
  }
  childPrint() {
    super.print()
  }
}
const child3 = new Child3()
child3.childPrint() // child; 注意
// 注意：使用super要么作为函数调用，要么当作一个对象并且访问他的属性或方法，一定不能只是用super不做任何操作

// prototype, __proto__
const obj = new Object()
console.log(obj.__proto__ === Object.prototype) // true
// 子类的__proto__指向的是父类本身
// 子类的prototype属性的__proto__指向的是父类的prototype属性

# 原生构造函数的继承

// ES5中原生构造函数是没法继承的，ES6中允许
class CostomArray extends Array {
  constructor(...args) {
    super(...args)
  }
}
const arr = new CostomArray(3)
arr.fill('#')
console.log(arr) // ['#', '#', '#']
console.log(arr.join('_')) // #_#_#
const arr2 = new CostomArray(3, 4, 5)
console.log(arr2) // [3, 4, 5]

// ES5中的继承是先创建子构造函数的this，然后再将父构造函数的方法、属性添加到this中
// ES6的继承是先从父类取到实例对象this，然后在调用super函数之后，再将子类的属性和方法加到this上
