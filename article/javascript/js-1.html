# Symbol

表示一种独一无二的值

# 基本用法

const s1 = Symbol()
console.log(s1) // Symbol()
const s2 = Symbol()
console.log(s2) // Symbol()
// console.log(s1 === s2) // error:This condition will always return 'false' since the types 'unique symbol' and 'unique symbol' have no overlap.

# 传参

// 可以传参
const s3 = Symbol('terry')
console.log(s3) // Symbol(terry)
// 这个字符串是给s3做一个标识用
const s4 = Symbol('terry')
console.log(s4)
// console.log(s3 === s4) // error: 错误同上
// 我们传入的字符串只是一个标识，只是为了后期打印看到是哪个标识创建的值，
// 所以只能用字符串或者可以转为字符串的类型如number
// 同标识的值彼此之间是不相等的
// 好比同名不同人

// symbol值是不能和其它类型的值做运算的
// s4 + 12 // error: 运算符“+”不能应用于类型“unique symbol”和“12”

// symbol值可以转换为字符串或布尔类型的值
console.log(s4.toString()) // 'Symbol(terry)'
console.log(Boolean(s4)) // true
// 可以隐式转换
console.log(!s4) // false

# 作为属性名

// 作为属性名
// ES特性
let prop = 'name'
const infoObj1 = {
  // name: 'terry' // {name: "terry"}
  [prop]: 'terry' // {name: "terry"}
}
console.log(infoObj1)
const s5 = Symbol('name')
const infoObj2 = {
  [s5]: 'terry'
}
console.log(infoObj2) // {Symbol(name): "terry"}
// 好处： 由于Symbol值是独一无二的，所以当用Symbol值做属性名时，该属性会保证不会被别的变量或者属性名覆盖
infoObj2[s5] = 'young'
console.log(infoObj2) // {Symbol(name): "young"}
// infoObj2.s5 = 'young' // error:类型“{ [s5]: string; }”上不存在属性“s5”
// 一般属性值可以通过infoObj2.xx或者infoObj2['xx'],symbol必须通过infoObj2[xx(symbol值)]赋值

// 属性的遍历
// 当Symbol值作为属性名时，以下几种是不能遍历到Symbol作为属性名的属性
const infoObj3 = {
  [s5]: 'terry',
  age: 22,
  sex: 'man'
}
for (const key in infoObj3) {
  console.log(key)
}
// age sex
console.log(Object.keys(infoObj3)) // ["age", "sex"]
console.log(Object.getOwnPropertyNames(infoObj3)) // ["age", "sex"]
console.log(JSON.stringify(infoObj3)) // "{"age":22,"sex":"man"}"
// 可以获取到的方法
console.log(Object.getOwnPropertySymbols(infoObj3)) // [Symbol(name)]
console.log(Reflect.ownKeys(infoObj3)) // ["age", "sex", Symbol(name)]
// Symbol值作为属性名可以在ES6中实现类的私有属性和私有方法的效果

# Symbol 的静态方法 for()、keyFor()

// Symbol的静态方法for()、keyFor()
const s6 = Symbol.for('terry')
const s7 = Symbol.for('terry')
// console.log(s6 === s7) // 在js中为true
// Symbol.for方法会在全局找有没有一样标识的Symbol值，没有的话会创建一个，有的话就直接赋值
console.log(Symbol.keyFor(s6)) // terry
// Symbol.keyFor方法会返回Symbol.for方法创建的值
