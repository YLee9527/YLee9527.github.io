# 泛型

使用 TS 最主要的原因是因为它可以很好地约束 API 使用者来规范的使用插件或库，这要求我们的 API 设定既要全面又要考虑可承接性，因此可以在 TS 中使用泛型，来让 API 支持多种类型的数据，同时又很好的支持类型结构的审查。

# 基本使用

// 简单场景
const getArrayA = (value: any, times: number = 5): any[] => {
  return new Array(times).fill(value)
}
getArrayA(2) // [2,2,2,2,2]
getArrayA(2, 3) // [2,2,2]
getArrayA(2, 3).map(item => item.length) // 打印undefined 应该要报错，但是没有（丢失类型的检测）

// 使用泛型约束该函数类型
// 此处的T就是泛型变量，代表固定的一种类型
const getArrayB = <T>(value: T, times: number = 5): T[] => {
  return new Array(times).fill(value)
}
// getArrayB<number>(2, 3).map((item) => item.length)  // error: 类型“number”上不存在属性“length”

# 泛型变量

// 泛型变量
// 此处的T不是固定名，可以换成任意写法，一般一个大写字母

// 复杂案例
const getArrayC = <T, U>(
  param1: T,
  param2: U,
  times: number
): Array<[T, U]> /*返回元组类型*/ => {
  return new Array(times).fill([param1, param2])
}
// 调用该方法没有明确传入类型，这是TS编辑器会根据我们的传参自动推断出类型
console.log(getArrayC(1, 'a', 3)) // [[1, 'a'], [1, 'a'], [1, 'a']]
getArrayC(1, 'a', 3).forEach(item => {
  console.log(item[0]) // 1 1 1
  // console.log(item[0].length)// error: 类型“number”上不存在属性“length”
})

# 泛型在类型定义中的使用

// 泛型在类型定义中的使用
// 定义函数类型
// const getArrayD: <T>(arg: T, times: number = 5) => T[]  // error: 只允许在函数或构造函数实现中使用参数初始化表达式。
let getArrayD: <T>(arg: T, times: number) => T[]
// 函数体
getArrayD = (arg: any, times: number) => {
  return new Array(times).fill(arg)
}
// getArrayD(123, 2).map((item) => item.length) // error: 类型“number”上不存在属性“length”。
// 类型别名
type GetArray = <T>(arg: T, times: number) => T[]
const getArrayE: GetArray = (arg: any, times: number) => {
  return new Array(times).fill(arg)
}
// 注意类型别名和函数类型的区别
// 接口
// interface GetArrayA {
//     <T>(arg: T, times: number): T[]
// }
// 将泛型变量提升到接口最外层，接口里面定义的所有类型都可以使用泛型变量（之前计数器的例子）
interface GetArrayB<T> {
  (arg: T, times: number): T[]
  array: T[]
}

# 泛型约束

泛型约束：对泛型变量的条件限制；

我们在定义一个类型结构时，可以使用泛型变量在多个地方保证多个值是同一种类型，或者有这个类型构成的其他元素

// 泛型约束：对泛型变量的条件限制
// 我们在定义一个类型结构时，可以使用泛型变量在多个地方保证多个值是同一种类型，或者有这个类型构成的其他元素
interface ValueWithLength {
  length: number
} // 必须要有length属性
const getArrayF = <T extends ValueWithLength>(arg: T, times: number): T[] => {
  return new Array(times).fill(arg)
}
getArrayF([1, 2], 3)
getArrayF('12', 3)
getArrayF(
  {
    length: 2
  },
  3
)
getArrayF(12, 3) // error: 类型“12”的参数不能赋给类型“ValueWithLength”的参数。

// 在泛型约束中使用类型参数
const getProps = (obj, propName) => {
  return obj[propName]
}
const objs = {
  a: 1,
  b: 2
}
getProps(objs, 'a') // 1
getProps(objs, 'c') // 返回undefined，应该报错，使在编译阶段就应该意识到错误
// keyof 索引类型， K相当于一个联合类型(a或者b)
const getPropsA = <T, K extends keyof T>(obj: T, propName: K) => {
  return obj[propName]
}
// getPropsA(objs, 'c') // error: 类型“"c"”的参数不能赋给类型“"a" | "b"”的参数。
