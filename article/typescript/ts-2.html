# 接口

接口能让我们为我们的代码定义一些结构，让其他人使用我们工具或者函数时，遵循一些规定

# 基本用法

// 基本用法
// 使用接口（简单直观）
interface NameInfo {
  firstName: string
  lastName: string
}
const getFullName = ({ firstName, lastName }: NameInfo): string => {
  return `${firstName} ${lastName}`
}
// 如果不使用接口，则需要这样
// const getFullName2 = ({
//   firstName,
//   lastName,
// }: {
//   firstName: string
//   lastName: string,
// }): string => {
//   return `${firstName} ${lastName}`
// }
getFullName({
  firstName: 'Young',
  lastName: 'Terry'
})
// getFullName({
//     firstName: 'Young',
//     lastName: 996,
//   })  // error:不能将类型“number”分配给类型“string”

# 可选属性

// 可选属性
interface Vagetable {
  color?: string // ？表示可选属性
  type: string
}
const getVagetables = ({ color, type }: Vagetable) => {
  return `A ${color ? color + ' ' : ''}${type}`
}
getVagetables({
  type: 'tomato'
})

// 多余属性检查
// getVagetables({
//     type: 'tomato',
//     color: 'red',
//     size: 99
//   })  // error:“size”不在类型“Vagetable”中
// 解决方法（绕开多于属性检测）
// 1、类型断言
getVagetables({
  type: 'tomato',
  color: 'red',
  size: 99
} as Vagetable)
// 2、接口使用索引签名,在保证该有的有的前提下，可以传入任意多属性
interface Vagetable {
  color?: string
  type: string
  [prop: string]: any
}
// 3、利用类型兼容性,如果想把a赋值给b，那么a必须要有b里面的属性，多了无所谓。
const vagetablesInfo = {
  type: 'tomato',
  color: 'red',
  size: 99
}
getVagetables(vagetablesInfo)

# 只读属性

// 只读属性
interface Vagetable2 {
  color?: string
  readonly type: string // 表示只读属性
}
let vagetablesObj: Vagetable2 = {
  type: 'tomato'
}
// vagetablesObj.type = 'carrot' // error: type为只读属性
interface ArrayInter {
  0: number
  readonly 1: string
}
// let arr: ArrayInter = ['1', '2'] // error: 不能将类型“string”分配给类型“number”
let arr: ArrayInter = [1, '2']
// arr[1] = '3' // error:索引号1为只读类型

  0: 'super_admin',

# 接口的继承

// 接口的继承
// 和类的继承相似，可以提高接口复用性
interface Vagetables {
  color: string
}
interface Tomato extends Vagetables {
  radius: number
}
// 相当于：
// interface Tomato {
//   color: string
//   radius: number
// }

// 混合类型接口
interface Counter {
  (): void
  count: number
}
const getCounter = (): Counter => {
  const c = () => {
    c.count++
  }
  c.count = 0
  return c
}
const couter: Counter = getCounter()
couter()
console.log(couter.count) // 1
couter()
console.log(couter.count) // 2
couter()
console.log(couter.count) // 3
