# 基础类型

包括十一个基础类型和一个类型断言

  • 布尔值
  • 数值
  • 字符串
  • 数组
  • 元组
  • 枚举值
  • any
  • void
  • null、undefined
  • never
  • object
  • 类型断言
/**********布尔类型***********/
let bool1: boolean = false
let bool2: boolean
bool2 = false
// 赋值给其他类型会报错
// bool1 = 123    error:不能将类型“123”分配给类型“boolean”

/**********数值类型***********/
// （TS和JS一样所有的数字都是浮点类型(number类型)）
let num: number = 233
// 支持二进制、八进制、十进制和十六进制字面量
num = 0b101011
num = 0o156
num = 0x99b
// 其他特性同布尔类型

/**********字符串类型***********/
let str: string
str = 'terry'
str = `它的大小为${num}`
console.log(str) // 它的大小为2459
// 其他特性同布尔类型

/**********数组类型***********/
// 针对特定类型的数组
let arr1: number[]
// 或者写成：let arr1: Array<number>
arr1 = [1, 3]
// arr1 = [1,3, '5']   error: 不能将类型“string”分配给类型“number”
let arr2: (string | number)[]
// 或者写成 let arr2: Array<string|number>
arr2 = [1, 3, '5']

/**********元组类型***********/
// 相当于数组的扩展，不同点在于：元组需要固定长度，固定每个位置上的类型
let tuple: [string, Array<number>, string[], number]
// tuple = [9, 5, 2, 7]  error
// tuple = ['9', [5], ['2'], 5]  error: 属性“length”的类型不兼容(2.6版本之前可以出现越界元素)
tuple = ['9', [5], ['2'], 7]

/**********枚举类型***********/
enum Roles {
  SUPER_ADMIN,
  ADMIN,
  USER
}
console.log(Roles.SUPER_ADMIN) // 0
console.log(Roles.ADMIN) // 1
console.log(Roles.USER) // 2
console.log(Roles[0]) //SUPER_ADMIN
console.log(Roles[1]) // ADMIN
console.log(Roles[2]) // USER
// 有时候服务端返回给的权限不是名字而是0、1、2这种，使用枚举就可以更直观知道具体权限是什么
// if(roles === Roles.SUPER_ADMIN) {
//     ...
// }
//
// 可以指定
// enum Roles {
//     SUPER_ADMIN,
//     ADMIN = 7,
//     USER
//   }
//   console.log(Roles.SUPER_ADMIN) // 0
//   console.log(Roles.ADMIN) // 7
//   console.log(Roles.USER) // 8

// 该枚举解析成JS为
// var Roles;
// (function (Roles) {

//     Roles[Roles["ADMIN"] = 1] = "ADMIN";
//     Roles[Roles["USER"] = 2] = "USER";
// })(Roles || (Roles = {}));

/**********any类型***********/
// 可以赋任何类型的值
let anyValue: any
anyValue = 998
anyValue = '998'
anyValue = [998]
// ...
// 注意定义const常量时，必须有初始化值，不像let变量
const anyArray: Array<any> = ['9', [5], ['2'], 7]

/**********void类型***********/
// 表示什么类型都不是，一般定义一个函数并且没有返回值时，那么指定函数的返回值类型就为void
const consoleText = (text: string | number): void => {
  console.log(text)
  // 该函数没有任何return的返回，指定void类型
}
// consoleText([9,3,4])  error:类型“number[]”的参数不能赋给类型“string | number”的参数
let v: void
v = undefined
// v = null  // 默认会报错，需要修改tsconfig的配置

/**********null和undefined类型***********/
// 既是值也是类型
let u: undefined
u = undefined
let n: null
n = null

/**********never类型***********/
// 情况一：报错
const errorFunc = (message: string): never => {
  throw new Error(message)
}
// 情况二：死循环
const infiniteFunc = (): never => {
  while (true) {}
}

/**********object类型***********/
// 引用类型
let obj: object = {
  name: 'terry'
}
const objFunc = (obj: object): void => {
  console.log(obj)
}

/**********类型断言***********/
// 排除TS类型检查，强制定义类型
// const getLength = (target:string|number):number => {
//     if(target.length || target.length === 0) {
//         return target.length
//     } else {
//         return target.toString().length
//     }
// }
// error: 类型“string | number”上不存在属性“length”。类型“number”上不存在属性“length”。
// 使用类型断言(两种写法)
const getLength = (target: string | number): number => {
  if ((<string>target).length || (target as string).length === 0) {
    return (<string>target).length
  } else {
    return target.toString().length
  }
}
