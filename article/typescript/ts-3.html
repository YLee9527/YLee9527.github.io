# 函数

# 定义函数类型

// 为函数定义类型
// 使用ES5的写法
function addFuncA(arg1: number, arg2: number): number {
  return arg1 + arg2
}
// 箭头函数
const addFuncB = (arg1: number, arg2: number) => arg1 + arg2

// 封装函数和插件时，会提供回调函数，有时会对回调函数有一定要求（参数、返回值等）
// 定义函数类型
let addFuncC: (x: number, y: number) => number
addFuncC = (arg1: number, arg2: number): number => arg1 + arg2
// addFuncC = (arg1: string, arg2: number) => arg1 + arg2  // error: 参数“arg1”和“x” 的类型不兼容。不能将类型“number”分配给类型“string”。
// 注意：
// 函数中使用函数体之外定义的变量，该变量的类型不体现在这个函数类型定义中,如果定义返回值类型，还是要注意该变量类型
let arg3 = 3
addFuncC = (arg1: number, arg2: number) => arg1 + arg2 + arg3

// 使用接口定义函数类型，tslint会推荐使用类型别名

// 适用类型别名定义类型
type Add = (x: number, y: number) => number // Add类型等同于定义的函数类型
type isString = string // 此时isString类型等同于string
let addFuncD: Add // 指定函数类型
addFuncD = (arg1: number, arg2: number) => arg1 + arg2

addFuncF = (x: number, y: number) => x + y

# 剩余参数

// 剩余参数
// ES5
function handleDataA() {
  if (arguments.length === 1) {
    return arguments[0] * 2
  } else if (arguments.length === 2) {
    return arguments[0] * arguments[1]
  } else {
    return Array.prototype.slice.apply(arguments).join('_')
  }
}
// handleDataA(2)  // 4
// handleDataA(2, 3)  // 6
// handleDataA(2,3,3) // '2_3_3'
// ES6
const handleDataB = (...args) => {
  console.log(args)
}
// handleDataB(2, 3)  // [2, 3]
// ...操作符：可以拆解一些数组或对象
// TS中
const handleDataC = (arr1: number, ...args: number[]) => {
  // ...
}

# 重载

// 重载
// TS中的重载时允许我们使用（只能使用）function定义好几个函数，然后传入不同个数不同参数的参数，会自动判断出实际应该返回的结果
function handleDataD(x: string): string[] // 函数重载
function handleDataD(x: number): number[] // 函数重载
// 函数实体（不是重载的一部分）
function handleDataD(x: any): any {
  if (typeof x === 'string') {
    return x.split('')
  } else {
    return x.toString.split('').map((item: string) => Number(item))
  }
}
handleDataD('123') // ['1', '2', '3']  //function handleDataD(x: string): string[] (+1 overload)
handleDataD(123) // [1, 2, 3]  // function handleDataD(x: number): number[] (+1 overload)
// 以上就是函数的重载，利用这一特性，编译器会帮助我们，根据我们传入不同类型或个数的参数，判断我们返回的是什么类型
handleDataD('234').map(item => {
  return item.toFixed()
}) // error: 属性“toFixed”在类型“string”上不存在
handleDataD(234).map(item => {
  return item.length
}) // error: 类型“number”上不存在属性“length”
