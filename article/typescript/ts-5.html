# TS 中定义类

// TS中定义类
class Point {
  public x: number
  public y: number
  constructor(x: number, y: number) {
    this.x = x
    this.y = y
  }
  public getPosition() {
    return `${this.x}, ${this.y}`
  }
}
const point = new Point(3, 8)
console.log(point) // Point {x: 3, y: 8}
class ChildPoint extends Point {
  constructor(x: number, y: number) {
    super(x, y)
  }
}

# 修饰符

  • public：表示公共的，用来创建实例后可以用来访问的，也就是类定义的外部可以使用和访问的属性和方法（默认）
  • private：表示私有的， 他修饰的属性在类的定义外面是无法访问的
  • protected: 表示受保护的，用它修饰的成员在继承该类的子类中可以访问
  • readonly：将属性设置为只读,并且还有指定一个其他的修饰符
// 修饰符
// public：表示公共的，用来创建实例后可以用来访问的，也就是类定义的外部可以使用和访问的属性和方法（默认）
// private： 表示私有的， 他修饰的属性在类的定义外面是无法访问的
class Parent {
  private age: number
  constructor(age: number) {
    this.age = age
  }
}
const p = new Parent(20)
console.log(p) // Parent {age: 20}
// console.log(p.age) // error: Property 'age' is private and only accessible within class 'Parent'.
// console.log(Parent.age) // error: 类型“typeof Parent”上不存在属性“age”
class Child extends Parent {
  constructor(age: number) {
    super(age)
    // console.log(super.age) // error: 通过 "super" 关键字只能访问基类的公共方法和受保护方法。ts(2340)
  }
}
// protected: 表示受保护的，用它修饰的成员在继承该类的子类中可以访问
class Parent {
  protected age: number
  constructor(age: number) {
    this.age = age
  }
  protected getAge() {
    return this.age
  }
}
const p = new Parent(20)
console.log(p) // Parent {age: 20}
// console.log(p.age) // error: 属性“age”受保护，只能在类“Parent”及其子类中访问
// console.log(Parent.age) // error: 类型“typeof Parent”上不存在属性“age”,
class Child extends Parent {
  constructor(age: number) {
    super(age)
    // console.log(super.age) // 通过 "super" 关键字只能访问基类的公共方法和受保护方法。
    console.log(super.getAge()) // 20
  }
}
// 注意 protected可以用来修饰constructor构造函数，此时类就不能用来创建实例，只能被子类继承

// readonly：将属性设置为只读,并且还有指定一个其他的修饰符
class UserInfo {
  public readonly name: string
  constructor(name: string) {
    this.name = name
  }
}
const userInfo = new UserInfo('terry')
console.log(userInfo) // UserInfo {name: "terry"}
// userInfo.name = 'young' // error: Cannot assign to 'name' because it is a read-only property

# 参数属性、静态属性、可选类属性

// 参数属性: 在constructor传参中加入属性，简化过程
class A {
  constructor(public name: string) {
    this.name = name
  }
}

// 静态属性 static
class B {
  public static age: number = 18
  private static getAge() {
    return B.age
  }
  constructor() {}
}
const b = new B()
// console.log(b.age) // error: Property 'age' is a static member of type 'B'
console.log(B.age) // 18
// console.log(B.getAge) // error: 属性“getAge”为私有属性，只能在类“B”中访问。

// 可选类属性： ?
class Info {
  public name: string
  public age?: number
  constructor(name: string, age?: number) {
    this.name = name
    this.age = age
  }
}
// const info = new Info() // error：应有 1-2 个参数，但获得 0 个
const info = new Info('terry')
const info2 = new Info('terry', 20)
console.log(info) // Info {name: "terry", age: undefined}
console.log(info2) // Info {name: "terry", age: 20}

# 存取器

存取器（之前 ES6 的存值函数和取值函数），在设置属性值时候调用的函数，和访问属性值的时候调用的函数

// 存取器（之前ES6的存值函数和取值函数），在设置属性值时候调用的函数，和访问属性值的时候调用的函数
class Info2 {
  public name: string
  public age?: number
  private infoStr: string = '666'
  constructor(name: string, age?: number) {
    this.name = name
    this.age = age
  }
  get infoStrFn() {
    return this.infoStr
  }
  set infoStrFn(value) {
    console.log(`setter: ${value}`)
    this.infoStr = value
    console.log(this.infoStr)
  }
}
const info4 = new Info2('terry', 20)
info4.infoStrFn = 'terry: 20'
console.log(info4.infoStrFn) // setter: terry: 20
// 和ES6没什么区别

# 抽象类与非抽象类

// 抽象类： 一般用来被其他类继承而不直接创建实例，使用abstract关键字
abstract class People {
  constructor(public name: string) {}
  // 定义一个抽象方法：不需要具体方法，有返回值
  public abstract printName(): void
}
// const p1 = new People() // error: 无法创建抽象类的实例。
class Man extends People {
  constructor(name: string) {
    super(name)
    this.name = name
  }
  // 非抽象类“Man”不会实现继承自“People”类的抽象成员“printName”,所以要手动实现一下
  public printName() {
    console.log(this.name)
  }
}
const man = new Man('terry')
// abstract还能定义属性和存取器
abstract class People2 {
  public abstract name: string
  abstract get insideName(): string
  abstract set insideName(value: string) // 存值器函数不能标记返回值类型
}

// 实例属性
const man2: Man = new Man('terry')

# 类类型接口与接口继承类

// 类类型接口(是使用该接口的类定义的实例)
interface FoodInterface {
  type: string
}
class FoodClass implements FoodInterface {
  // static type: string  // error: 类“FoodClass”错误实现接口“FoodInterface”。
  public type: string = ''
}
// 接口继承类（会继承这个类的成员（包括private和protected修饰的(需要继承该类)）及类型，但是不包括其实现）
class AB {
  protected name: string = ''
}
interface I extends AB {}
// class BA implements I {
//   public name: string = ''
// }  // error: 属性“name”受保护，但类型“BA”并不是从“AB”派生的类。
class BA extends AB implements I {
  public name: string = ''
}
