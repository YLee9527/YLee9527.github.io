# 类型兼容性

// 类型兼容性
interface Infos {
  name: string
}
let a: Infos
const b = { name: 'terry' }
const c = { age: 18 }
const d = {
  name: 'terry',
  age: 18
}
a = b
// a = c // error: 'name' is declared here.
a = d
// 赋值的必须有被赋值的所有字段，多了无所谓，这种检测是递归检测的，即使深层次的嵌套也可以检测到

# 函数的兼容性

// 函数的兼容性
// 参数个数
let x = (a: number) => 0
let y = (b: number, c: number) => 0
// x = y // error 不能将类型“(b: number, c: number) => number”分配给类型“(a: number) => number”
y = x
const arrs = [1, 2, 3]
arrs.forEach((item, index, array) => {
  console.log(item)
})
// 可以写成（可以小于等于被赋值的参数个数）
arrs.forEach(item => {
  console.log(item)
})

// 参数类型
let x1 = (a: number) => 0
let y1 = (b: string) => 0
// x1 = y1 // error: 参数“b”和“a” 的类型不兼容。

// 可选参数和剩余参数
let getSum = (
  arr: number[],
  callback: (...args: number[]) => number
): number => {
  return callback(...arr)
}
const res = getSum(
  [1, 2, 3],
  (...args: number[]): number => args.reduce((a, b) => a + b, 0)
)
console.log(res) // 6
const res2 = getSum(
  [1, 2, 3],
  (arg1: number, arg2: number): number => arg1 + arg2
)
console.log(res2) // 3
// 当要被赋值的函数参数中包含剩余参数时，赋值的函数可以用任意个数参数替代，但是类型需要对应
// 剩余参数可以看作是无数可选参数

// 函数参数双向协变(可以通过配置关掉)
let funcA = (arg: number | string): void => {}
let funcB = (arg: number): void => {}
// funcA = funcB （开启双向协变则不会报错，否则报错）
funcB = funcA

// 返回值类型
let x2 = (): string | number => 0
let y2 = (): string => 'a'
x2 = y2
// y2 = x2  // error:不能将类型“number”分配给类型“string”

# 函数重载

要 function 关键字定义

// 函数重载（要function关键字定义）
function merge(arg1: number, arg2: number): number
function merge(arg1: string, arg2: string): string
function merge(arg1: any, arg2: any) {
  return arg1 + arg2
}
merge(1, 2) // function merge(arg1: number, arg2: number): number (+1 overload)
merge('1', '2') // function merge(arg1: string, arg2: string): string (+1 overload)
function sum(arg1: number, arg2: number): number
function sum(arg1: any, arg2: any) {
  return arg1 + arg2
}
let func = merge
// func = sum // error:参数“arg1”和“arg1” 的类型不兼容。(两种类型对一种类型，不兼容)

enum StatusInter {
  On,
  Off
}
enum AminalInter {
  Dog,
  Cat

let s = StatusInter.On
s = 3
// 数字枚举类型不与其他数字枚举类型兼容

# 类的兼容性

// 类的兼容性
// 只比较实例的成员，类的静态成员和构造函数不进行比较
class AminalClass {
  public static age: number
  constructor(public name: string) {}
}
class PeopleClass {
  public static age: string
  constructor(public name: string) {}
}
class FoodClass {
  constructor(public name: number) {}
}
// 当使用类作为类型指定一个变量他的类类型的时候，其实检测的是类的实例
let animal: AminalClass
let peoples: PeopleClass
let food: FoodClass
// animal = peoples
// animal = food // error:不能将类型“FoodClass”分配给类型“AminalClass”
